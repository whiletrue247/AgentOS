04_Engine — Soul Evolver (Memory → SOUL 自動畢業)

定期掃描 Memory 中高頻/高重要性的事實記憶， 當滿足畢業條件時，將知識「畢業」到 SOUL.md 的 Learned Patterns 區塊。

設計原則：

  • 規則式畢業（importance ≥ threshold 且 access ≥ N）
  • 首次實作必須有人類確認門檻（防止噪音污染 SOUL）
  • SOUL.md 備份機制（.bak）
  • 可由 LoRATunerSchedule 或獨立排程呼叫

靈感來源：Voyager (Minecraft Agent) 的 Skill Library 自我進化機制。

 50class SoulEvolver:
 51    """
 52    Memory → SOUL 自動畢業引擎。
 53    
 54    掃描高頻/高重要性記憶，經人類確認後追加到 SOUL.md。
 55    
 56    Args:
 57        soul_path: SOUL.md 檔案路徑
 58        memory_manager: MemoryManager 實例
 59        importance_threshold: 重要性門檻 (0.0~1.0)
 60        min_access_count: 最少存取次數
 61        auto_graduate: 是否自動畢業（無人類確認）
 62        max_candidates_per_cycle: 每次最多處理幾筆候選
 63    """
 64
 65    def __init__(
 66        self,
 67        soul_path: str = "SOUL.md",
 68        memory_manager: Any = None,
 69        lora_tuner: Any = None,
 70        importance_threshold: float = 0.8,
 71        min_access_count: int = 5,
 72        auto_graduate: bool = False,
 73        max_candidates_per_cycle: int = 5,
 74    ):
 75        self.soul_path = Path(soul_path)
 76        self._memory = memory_manager
 77        self._lora_tuner = lora_tuner  # LoRATunerSchedule 實例
 78        self.importance_threshold = importance_threshold
 79        self.min_access_count = min_access_count
 80        self.auto_graduate = auto_graduate
 81        self.max_candidates_per_cycle = max_candidates_per_cycle
 82
 83        # 畢業歷史紀錄 (防止重複畢業)
 84        self._graduated_ids: set[str] = set()
 85        self._load_graduated_ids()
 86
 87        logger.info(
 88            f"🎓 SoulEvolver 初始化: threshold={importance_threshold}, "
 89            f"min_access={min_access_count}, auto={auto_graduate}, "
 90            f"lora={'connected' if lora_tuner else 'none'}"
 91        )

 94    # 掃描候選
 95    # ========================================
 96
 97    async def scan_candidates(self) -> List[GraduationCandidate]:
 98        """
 99        掃描 Memory 中符合畢業條件的記憶。
100        
101        Returns:
102            符合條件的畢業候選列表
103        """
104        if not self._memory:
105            logger.warning("⚠️ SoulEvolver: 未注入 MemoryManager，無法掃描")
106            return []
107
108        candidates: List[GraduationCandidate] = []
109
110        try:
111            # 搜尋高重要性的事實型記憶
112            items = await self._memory.search(
113                query="*",
114                top_k=50,
115                min_importance=self.importance_threshold,
116                content_type="fact",
117            )
118
119            for item in items:
120                # 跳過已畢業的記憶
121                if item.memory_id in self._graduated_ids:
122                    continue
123
124                # 檢查存取次數
125                access_count = item.metadata.get("access_count", 0)
126                if access_count < self.min_access_count:
127                    continue
128
129                candidates.append(GraduationCandidate(
130                    memory_id=item.memory_id,
131                    content=item.content,
132                    content_type=item.content_type,
133                    importance=item.importance,
134                    access_count=access_count,
135                    created_at=item.t_created,
136                    source=item.metadata.get("source", "memory"),
137                ))
138
139        except Exception as e:
140            logger.error(f"❌ SoulEvolver scan 失敗: {e}")
141
142        # 按重要性排序，取前 N 個
143        candidates.sort(key=lambda c: (c.importance, c.access_count), reverse=True)
144        candidates = candidates[:self.max_candidates_per_cycle]
145
146        logger.info(f"🔍 掃描到 {len(candidates)} 個畢業候選")
147        return candidates

153    @staticmethod
154    def format_graduation(candidates: List[GraduationCandidate]) -> str:
155        """
156        將候選記憶格式化為 SOUL.md 的 Learned Patterns 格式。
157        
158        Args:
159            candidates: 畢業候選列表
160            
161        Returns:
162            格式化後的 markdown 文字
163        """
164        if not candidates:
165            return ""
166
167        lines = [
168            "",
169            f"<!-- Graduated by SoulEvolver at {datetime.now().strftime('%Y-%m-%d %H:%M')} -->",
170        ]
171
172        for c in candidates:
173            # 格式: - [Learned] 內容 (importance: X, accessed: N times)
174            lines.append(
175                f"- **[Learned]** {c.content} "
176                f"_(importance: {c.importance:.2f}, accessed: {c.access_count}x, "
177                f"source: {c.source})_"
178            )
179
180        return "\n".join(lines)

229    # ========================================
230    # 執行畢業 (寫入 SOUL.md)
231    # ========================================
232
233    def graduate(self, candidates: List[GraduationCandidate]) -> int:
234        """
235        將候選記憶寫入 SOUL.md。
236        
237        Args:
238            candidates: 已批准的畢業候選
239            
240        Returns:
241            成功畢業的數量
242        """
243        if not candidates:
244            return 0
245
246        # 1. 備份 SOUL.md
247        if self.soul_path.exists():
248            backup_path = self.soul_path.with_suffix(".md.bak")
249            shutil.copy2(str(self.soul_path), str(backup_path))
250            logger.info(f"💾 SOUL.md 已備份至 {backup_path}")
251
252        # 2. 讀取現有內容
253        if self.soul_path.exists():
254            content = self.soul_path.read_text(encoding="utf-8")
255        else:
256            content = "# SOUL\n"
257
258        # 3. 尋找或建立 Learned Patterns 區塊
259        section_header = "## Learned Patterns"
260        if section_header not in content:
261            content += f"\n\n{section_header}\n"
262
263        # 4. 追加畢業內容
264        graduation_text = self.format_graduation(candidates)
265        
266        # 插入到 ## Learned Patterns 區塊的末尾
267        insert_pos = content.find(section_header) + len(section_header)
268        # 跳過 section header 後的換行
269        while insert_pos < len(content) and content[insert_pos] == '\n':
270            insert_pos += 1
271        
272        # 找到下一個 ## 區塊的位置（如果有的話）
273        next_section = content.find("\n## ", insert_pos)
274        if next_section == -1:
275            # 沒有下一個區塊，追加到末尾
276            content += graduation_text + "\n"
277        else:
278            # 在下一個區塊前插入
279            content = content[:next_section] + graduation_text + "\n" + content[next_section:]
280
281        # 5. 寫入
282        self.soul_path.write_text(content, encoding="utf-8")
283
284        # 6. 記錄已畢業的 ID
285        for c in candidates:
286            self._graduated_ids.add(c.memory_id)
287        self._save_graduated_ids()
288
289        logger.info(f"🎓 成功畢業 {len(candidates)} 條記憶到 SOUL.md")
290        return len(candidates)

293    # 完整進化週期
294    # ========================================
295
296    async def run_cycle(self) -> int:
297        """
298        執行一次完整的進化週期：掃描 → 預覽 → 畢業 → LoRA 微調。
299        
300        Returns:
301            本次畢業的記憶數量
302        """
303        logger.info("🔄 SoulEvolver: 開始進化週期")
304
305        # 1. 掃描候選
306        candidates = await self.scan_candidates()
307        if not candidates:
308            logger.info("ℹ️ 本次週期無畢業候選")
309            return 0
310
311        # 2. 人類確認（或自動批准）
312        approved = self.preview_changes(candidates)
313        if not approved:
314            logger.info("🚫 人類拒絕本次畢業")
315            return 0
316
317        # 3. 執行畢業
318        count = self.graduate(candidates)
319
320        # 4. 觸發 LoRA 微調（閉環）
321        if count > 0 and self._lora_tuner:
322            try:
323                logger.info(
324                    f"🧬 畢業完成 ({count} 條) → 觸發 LoRA 微調更新本地模型..."
325                )
326                await self._lora_tuner.trigger_tuning_job()
327                logger.info("✅ LoRA 微調已觸發，Self-Improvement 閉環完成")
328            except Exception as e:
329                logger.error(f"⚠️ LoRA 微調觸發失敗 (畢業仍有效): {e}")
330
331        return count

337    def _load_graduated_ids(self) -> None:
338        """從檔案載入已畢業的 memory_id 集合"""
339        id_file = self.soul_path.parent / ".soul_graduated_ids"
340        if id_file.exists():
341            try:
342                self._graduated_ids = set(
343                    id_file.read_text(encoding="utf-8").strip().splitlines()
344                )
345            except Exception:
346                self._graduated_ids = set()
347
348    def _save_graduated_ids(self) -> None:
349        """將已畢業的 memory_id 集合儲存到檔案"""
350        id_file = self.soul_path.parent / ".soul_graduated_ids"
351        try:
352            id_file.write_text(
353                "\n".join(sorted(self._graduated_ids)),
354                encoding="utf-8",
355            )
356        except Exception as e:
357            logger.error(f"⚠️ 無法儲存畢業 ID 列表: {e}")
