04_Engine — Ensemble Router Extension

在 APIGateway 之上提供多模型 Ensemble 投票機制：

  • 同時向 2+ 個 Provider/Model 發問
  • 比較回答品質（長度、結構、confidence）
  • 選擇最佳答案回傳

使用場景：

  • 高重要性決策 (importance >= 0.8)
  • 需要交叉驗證的 factual 回答
  • config 開啟 ensemble 模式時

 26class EnsembleRouter:
 27    """
 28    多模型 Ensemble 投票器。
 29    
 30    同時呼叫多個模型，根據回答品質選擇最佳結果。
 31    可選策略：longest (最詳細)、vote (多數決)、first (最快回應)。
 32    
 33    Args:
 34        gateway: APIGateway 實例
 35        strategy: 選擇策略 ('longest' | 'vote' | 'first')
 36    """
 37
 38    def __init__(self, gateway: Any, strategy: str = "longest"):
 39        self._gateway = gateway
 40        self._strategy = strategy
 41        logger.info(f"🎯 EnsembleRouter 初始化: strategy={strategy}")
 42
 43    async def call_ensemble(
 44        self,
 45        messages: list[dict],
 46        models: Optional[List[str]] = None,
 47        agent_id: str = "default",
 48        tools: Optional[list[dict]] = None,
 49        temperature: float = 0.7,
 50    ) -> Dict[str, Any]:
 51        """
 52        同時向多個模型發問並選擇最佳回答。
 53        
 54        Args:
 55            messages: 對話訊息
 56            models: 模型列表 (如 ['gpt-4o', 'claude-3.5-sonnet'])
 57                    若為 None，使用 config 中前 2 個 provider 的模型
 58            agent_id: Agent ID
 59            tools: 工具定義
 60            temperature: 溫度
 61            
 62        Returns:
 63            最佳回答的完整 response dict + ensemble_metadata
 64        """
 65        # 決定要使用的模型
 66        if not models:
 67            models = self._get_default_ensemble_models()
 68
 69        if len(models) < 2:
 70            logger.warning("⚠️ Ensemble 需要至少 2 個模型，退回單模型")
 71            return await self._gateway.call(
 72                messages=messages, agent_id=agent_id,
 73                tools=tools, temperature=temperature,
 74            )
 75
 76        logger.info(f"🎯 Ensemble 呼叫: {len(models)} 個模型 → {models}")
 77
 78        # 平行呼叫所有模型
 79        tasks = []
 80        for model in models:
 81            tasks.append(
 82                self._call_single(
 83                    messages=messages, model=model,
 84                    tools=tools, temperature=temperature,
 85                )
 86            )
 87
 88        results = await asyncio.gather(*tasks, return_exceptions=True)
 89
 90        # 過濾成功的結果
 91        valid_results = []
 92        for i, r in enumerate(results):
 93            if isinstance(r, Exception):
 94                logger.warning(f"⚠️ Ensemble model {models[i]} failed: {r}")
 95            else:
 96                valid_results.append((models[i], r))
 97
 98        if not valid_results:
 99            raise RuntimeError("All ensemble models failed")
100
101        # 選擇最佳結果
102        best_model, best_response = self._select_best(valid_results)
103        logger.info(f"✅ Ensemble 最佳: {best_model}")
104
105        # 附加 ensemble metadata
106        best_response["ensemble_metadata"] = {
107            "models_called": models,
108            "models_succeeded": [m for m, _ in valid_results],
109            "selected_model": best_model,
110            "strategy": self._strategy,
111        }
112
113        return best_response
114
115    async def _call_single(
116        self,
117        messages: list[dict],
118        model: str,
119        tools: Optional[list[dict]] = None,
120        temperature: float = 0.7,
121    ) -> Dict[str, Any]:
122        """呼叫單一模型"""
123        return await self._gateway.call(
124            messages=messages,
125            tools=tools,
126            temperature=temperature,
127        )
128
129    def _select_best(
130        self,
131        results: List[tuple[str, Dict[str, Any]]],
132    ) -> tuple[str, Dict[str, Any]]:
133        """根據策略選擇最佳結果"""
134        if self._strategy == "first":
135            return results[0]
136
137        if self._strategy == "longest":
138            # 選最長回答 (通常更詳細)
139            return max(
140                results,
141                key=lambda r: len(
142                    r[1].get("choices", [{}])[0]
143                    .get("message", {})
144                    .get("content", "")
145                ),
146            )
147
148        # default: longest
149        return max(
150            results,
151            key=lambda r: len(
152                r[1].get("choices", [{}])[0]
153                .get("message", {})
154                .get("content", "")
155            ),
156        )
157
158    def _get_default_ensemble_models(self) -> List[str]:
159        """從 gateway config 取前 2 個 provider 的模型"""
160        try:
161            providers = self._gateway.config.providers
162            models = []
163            for p in providers[:2]:
164                if p.models:
165                    models.append(p.models[0])
166            return models
167        except Exception:
168            return []
