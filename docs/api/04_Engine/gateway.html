 50    API 閘道器 (v5.0 SOTA)。

 55    def __init__(self, config: AgentOSConfig):
 56        self.config = config
 57        self._providers: dict[str, ProviderConfig] = {}
 58        self._call_history: list[APICallRecord] = []
 59
 60        for p in config.gateway.providers:
 61            self._providers[p.name] = p
 62            # 將 API key 注入 litellm 環境 (litellm 會自動讀取)
 63            self._register_provider_keys(p)
 64
 65        self._router = SmartRouter(config)
 66
 67        backend = "litellm" if LITELLM_AVAILABLE else "httpx-fallback"
 68        logger.info(f"🌐 API Gateway 啟動: {len(self._providers)} providers, backend={backend}, SmartRouter 就緒")
 69        
 70        if OTEL_AVAILABLE:
 71            self._tracer = trace.get_tracer(__name__)
 72            logger.info("🔭 OTLP Tracing 整合已啟用 (APIGateway)")
 73        else:
 74            self._tracer = None

 76    @staticmethod
 77    def _register_provider_keys(p: ProviderConfig) -> None:
 78        """將 provider 的 API key 注入 litellm 的環境變數映射。"""
 79        if not LITELLM_AVAILABLE or not p.api_key:
 80            return
 81        import os
 82        key_map = {
 83            "openai": "OPENAI_API_KEY",
 84            "anthropic": "ANTHROPIC_API_KEY",
 85            "google": "GEMINI_API_KEY",
 86            "deepseek": "DEEPSEEK_API_KEY",
 87            "groq": "GROQ_API_KEY",
 88            "mistral": "MISTRAL_API_KEY",
 89            "together": "TOGETHERAI_API_KEY",
 90        }
 91        env_var = key_map.get(p.name)
 92        if env_var:
 93            os.environ[env_var] = p.api_key

 95    def resolve_model(
 96        self,
 97        agent_id: str = "default",
 98        messages: list[dict] | None = None,
 99        tools: list[dict] | None = None,
100    ) -> tuple[ProviderConfig, str]:
101        """由 SmartRouter 決定最佳 Provider + Model。"""
102        if messages is None:
103            messages = []
104
105        provider_name, model, override_url = self._router.route(agent_id, messages, tools)
106        base_provider = self._providers.get(provider_name, ProviderConfig(name=provider_name))
107
108        call_provider = ProviderConfig(
109            name=base_provider.name,
110            api_key=base_provider.api_key,
111            base_url=override_url or base_provider.base_url,
112            models=base_provider.models,
113        )
114        return call_provider, model

143    async def call(
144        self,
145        messages: list[dict],
146        agent_id: str = "default",
147        tools: Optional[list[dict]] = None,
148        stream: bool = False,
149        temperature: float = 0.7,
150        **kwargs
151    ) -> dict[str, Any]:
152        """
153        發送 LLM API 請求，含自動重試與離線 failover。
154        優先使用 litellm；若 litellm 不可用則退回 httpx fallback。
155        """
156        # 防止 **kwargs 覆寫高危參數
157        for key in ["api_key", "api_base", "base_url", "organization"]:
158            kwargs.pop(key, None)
159
160        retry_cfg = self.config.engine.retry
161        last_error: Optional[Exception] = None
162
163        for attempt in range(retry_cfg.max_attempts):
164            provider, model = self.resolve_model(agent_id, messages, tools)
165
166            try:
167                if LITELLM_AVAILABLE:
168                    result = await self._call_via_litellm(
169                        provider=provider, model=model,
170                        messages=messages, tools=tools,
171                        stream=stream, temperature=temperature,
172                        **kwargs
173                    )
174                else:
175                    result = await self._call_via_httpx(
176                        provider=provider, model=model,
177                        messages=messages, tools=tools,
178                        temperature=temperature,
179                        **kwargs
180                    )
181                return result
182
183            except APIError as e:
184                last_error = e
185                if e.status_code == 599:
186                    logger.warning("🚨 Gateway 捕獲網路斷線 (599)，SmartRouter 切換離線模式！")
187                    self._router.set_offline_mode(True)
188                    continue
189
190                if e.status_code in retry_cfg.retryable_codes and attempt < retry_cfg.max_attempts - 1:
191                    wait = retry_cfg.backoff_multiplier ** attempt
192                    logger.warning(f"⚠️ API 錯誤 {e.status_code}，{wait}s 後重試 (attempt {attempt + 1})")
193                    await asyncio.sleep(wait)
194                else:
195                    raise
196
197        raise last_error or APIError(500, "Max retries exceeded")

200    # litellm backend

202    async def _do_call_via_litellm(
203        self,
204        provider: ProviderConfig,
205        model: str,
206        messages: list[dict],
207        tools: Optional[list[dict]] = None,
208        stream: bool = False,
209        temperature: float = 0.7,
210        **kwargs
211    ) -> dict[str, Any]:
212        """使用 litellm.acompletion() 進行 LLM 呼叫。"""
213        litellm_model = self._to_litellm_model(provider.name, model)
214
215        call_kwargs: dict[str, Any] = {
216            "model": litellm_model,
217            "messages": messages,
218            "temperature": temperature,
219            "stream": stream,
220        }
221
222        # Ollama 或其他自建端點需要 base_url
223        if provider.base_url:
224            call_kwargs["api_base"] = provider.base_url
225
226        # 如果有 API key 直接傳入 (覆蓋環境變數)
227        if provider.api_key:
228            call_kwargs["api_key"] = provider.api_key
229
230        # Tool calling
231        if tools:
232            call_kwargs["tools"] = [{"type": "function", "function": t} for t in tools]
233            # litellm 預設會自動選擇 tool_choice = "auto"
234
235        # 合併額外的 kwargs
236        call_kwargs.update(kwargs)
237
238        start = time.time()
239        try:
240            response = await litellm.acompletion(**call_kwargs)
241        except Exception as e:
242            error_str = str(e)
243            if "connect" in error_str.lower() or "timeout" in error_str.lower():
244                raise APIError(599, f"Network connection failed: {error_str}")
245            raise APIError(500, f"litellm error: {error_str}")
246
247        elapsed_ms = int((time.time() - start) * 1000)
248
249        # litellm 回傳 ModelResponse，轉為 dict
250        result = response.model_dump() if hasattr(response, "model_dump") else dict(response)
251
252        # 記錄呼叫
253        usage = result.get("usage", {})
254        input_tokens = usage.get("prompt_tokens", 0)
255        output_tokens = usage.get("completion_tokens", 0)
256
257        record = APICallRecord(
258            model=model,
259            provider=provider.name,
260            input_tokens=input_tokens,
261            output_tokens=output_tokens,
262            agent_id="default",
263        )
264        self._call_history.append(record)
265
266        logger.info(
267            f"✅ [{provider.name}/{model}] via litellm ({elapsed_ms}ms, "
268            f"in:{input_tokens} out:{output_tokens})"
269        )
270        return result
271            
272    async def _call_via_litellm(self, *args, **kwargs):
273        """帶有 OpenTelemetry 追蹤的 litellm 呼叫"""
274        if self._tracer:
275            with self._tracer.start_as_current_span(
276                "APIGateway.litellm_call",
277                attributes={
278                    "llm.provider": kwargs.get("provider", ProviderConfig(name="unknown")).name,
279                    "llm.model": kwargs.get("model", "unknown"),
280                    "llm.stream": kwargs.get("stream", False),
281                }
282            ) as span:
283                try:
284                    result = await self._do_call_via_litellm(*args, **kwargs)
285                    span.set_attribute("llm.usage.prompt_tokens", result.get("usage", {}).get("prompt_tokens", 0))
286                    span.set_attribute("llm.usage.completion_tokens", result.get("usage", {}).get("completion_tokens", 0))
287                    span.set_status(Status(StatusCode.OK))
288                    return result
289                except Exception as e:
290                    span.record_exception(e)
291                    span.set_status(Status(StatusCode.ERROR, str(e)))
292                    raise
293        else:
294            return await self._do_call_via_litellm(*args, **kwargs)

299    async def _call_via_httpx(
300        self,
301        provider: ProviderConfig,
302        model: str,
303        messages: list[dict],
304        tools: Optional[list[dict]] = None,
305        temperature: float = 0.7,
306        **kwargs
307    ) -> dict[str, Any]:
308        """Fallback: 用 httpx 直接打 OpenAI-compatible endpoint。"""
309        import httpx
310
311        base = provider.base_url or "https://api.openai.com"
312        endpoint = f"{base.rstrip('/')}/v1/chat/completions"
313
314        payload: dict[str, Any] = {
315            "model": model,
316            "messages": messages,
317            "temperature": temperature,
318            "stream": False,
319        }
320        if tools:
321            payload["tools"] = [{"type": "function", "function": t} for t in tools]
322        
323        # 合併額外的 kwargs
324        payload.update(kwargs)
325
326        headers: dict[str, str] = {"content-type": "application/json"}
327        if provider.api_key:
328            headers["authorization"] = f"Bearer {provider.api_key}"
329
330        start = time.time()
331        async with httpx.AsyncClient(timeout=120.0) as client:
332            try:
333                resp = await client.post(endpoint, json=payload, headers=headers)
334                resp.raise_for_status()
335                result = resp.json()
336            except httpx.ConnectError as e:
337                raise APIError(599, f"Network connection failed: {e}")
338            except httpx.HTTPStatusError as e:
339                raise APIError(e.response.status_code, f"API Error: {e.response.text}")
340
341        elapsed_ms = int((time.time() - start) * 1000)
342
343        input_tokens = result.get("usage", {}).get("prompt_tokens", 0)
344        output_tokens = result.get("usage", {}).get("completion_tokens", 0)
345
346        record = APICallRecord(
347            model=model, provider=provider.name,
348            input_tokens=input_tokens, output_tokens=output_tokens,
349            agent_id="default",
350        )
351        self._call_history.append(record)
352
353        logger.info(f"✅ [{provider.name}/{model}] via httpx ({elapsed_ms}ms, in:{input_tokens} out:{output_tokens})")
354        return result

356    def get_call_history(self) -> list[APICallRecord]:
357        """取得所有 API 呼叫紀錄 (供 Cost Guard 使用)"""
358        return self._call_history

144    async def call(
145        self,
146        messages: list[dict],
147        agent_id: str = "default",
148        tools: Optional[list[dict]] = None,
149        stream: bool = False,
150        temperature: float = 0.7,
151        **kwargs
152    ) -> dict[str, Any]:
153        """
154        發送 LLM API 請求，含自動重試與離線 failover。
155        優先使用 litellm；若 litellm 不可用則退回 httpx fallback。
156        """
157        # 防止 **kwargs 覆寫高危參數
158        for key in ["api_key", "api_base", "base_url", "organization"]:
159            kwargs.pop(key, None)
160
161        retry_cfg = self.config.engine.retry
162        last_error: Optional[Exception] = None
163
164        for attempt in range(retry_cfg.max_attempts):
165            provider, model = self.resolve_model(agent_id, messages, tools)
166
167            try:
168                if LITELLM_AVAILABLE:
169                    result = await self._call_via_litellm(
170                        provider=provider, model=model,
171                        messages=messages, tools=tools,
172                        stream=stream, temperature=temperature,
173                        **kwargs
174                    )
175                else:
176                    result = await self._call_via_httpx(
177                        provider=provider, model=model,
178                        messages=messages, tools=tools,
179                        temperature=temperature,
180                        **kwargs
181                    )
182                return result
183
184            except APIError as e:
185                last_error = e
186                if e.status_code == 599:
187                    logger.warning("🚨 Gateway 捕獲網路斷線 (599)，SmartRouter 切換離線模式！")
188                    self._router.set_offline_mode(True)
189                    continue
190
191                if e.status_code in retry_cfg.retryable_codes and attempt < retry_cfg.max_attempts - 1:
192                    wait = retry_cfg.backoff_multiplier ** attempt
193                    logger.warning(f"⚠️ API 錯誤 {e.status_code}，{wait}s 後重試 (attempt {attempt + 1})")
194                    await asyncio.sleep(wait)
195                else:
196                    raise
197
198        raise last_error or APIError(500, "Max retries exceeded")
