04_Engine — 主事件循環 (Engine)

AgentOS 的心臟。 把所有零件串在一起：Gateway → RateLimiter → Streamer → ToolSystem → Memory。

事件驅動架構：

  1. 收到 USER_MESSAGE
  2. 組裝 System Prompt (SOUL + Memory Context)
  3. 呼叫 LLM API (經 RateLimiter 節流)
  4. 若 LLM 回覆 tool_call → 交給 ToolSystem 執行 → 結果回傳 LLM → 重複
  5. 若 LLM 回覆文字 → 發送 AGENT_RESPONSE 事件 → 結束

 50def _is_retryable(error: Exception) -> bool:
 51    """
 52    判斷 API 錯誤是否可重試。
 53    可重試：Timeout、429 (Rate Limit)、5xx (Server Error)
 54    不可重試：401/403 (Auth)、400 (Bad Request)、其他未知錯誤
 55    """
 56    error_str = str(error).lower()
 57    error_type = type(error).__name__.lower()
 58
 59    # Timeout 類錯誤
 60    if "timeout" in error_str or "timeout" in error_type:
 61        return True
 62
 63    # 連線錯誤
 64    if "connection" in error_str or "connect" in error_type:
 65        return True
 66
 67    # HTTP status code 判斷
 68    status_code = getattr(error, "status_code", None)
 69    if status_code is not None:
 70        # 429 Rate Limit、5xx Server Error 可重試
 71        if status_code == 429 or 500 <= status_code < 600:
 72            return True
 73        # 4xx Client Error 不可重試
 74        if 400 <= status_code < 500:
 75            return False
 76
 77    # 檢查錯誤訊息中的 status code
 78    if "429" in error_str or "rate limit" in error_str:
 79        return True
 80    if any(f"{code}" in error_str for code in range(500, 504)):
 81        return True
 82
 83    # 預設不重試（保守策略）
 84    return False

199    async def handle_message(
200        self,
201        user_message: str,
202        agent_id: str = "default",
203        conversation_history: Optional[list[dict]] = None,
204    ) -> str:
205        """
206        處理一則使用者訊息，回傳 Agent 的最終回覆。
207        這是 Engine 的主入口。
208        """
209        self._step_count = 0
210        max_steps = self.config.engine.watchdog.max_steps
211
212        # 組裝對話歷史
213        messages = conversation_history or []
214
215        # 注入 System Prompt (SOUL + Memory Context)
216        system_prompt = self._soul_content or "You are a helpful AI Agent."
217
218        # 嘗試從 Memory 中取得相關上下文
219        if self._memory_manager:
220            try:
221                memory_ctx = await self._memory_manager.get_relevant_context(
222                    query=user_message, agent_id=agent_id
223                )
224                if memory_ctx:
225                    system_prompt += f"\n\n{memory_ctx}"
226            except Exception as e:
227                logger.warning(f"⚠️ Memory context 取得失敗: {e}")
228
229        # 確保 system message 在最前面
230        if not messages or messages[0].get("role") != "system":
231            messages.insert(0, {"role": "system", "content": system_prompt})
232        else:
233            messages[0]["content"] = system_prompt
234
235        # 加入使用者訊息
236        messages.append({"role": "user", "content": user_message})
237
238        # 發送 USER_MESSAGE 事件
239        await self.emit(EngineEvent(
240            event_type=EventType.USER_MESSAGE,
241            payload={"message": user_message, "agent_id": agent_id},
242        ))
243
244        # ReAct Loop
245        while self._step_count < max_steps:
246            self._step_count += 1
247            logger.info(f"🔄 ReAct Step {self._step_count}/{max_steps}")
248
249            # Rate limit
250            if self._rate_limiter:
251                estimated_tokens = 0
252                for m in messages:
253                    c = m.get("content")
254                    if isinstance(c, str):
255                        estimated_tokens += len(c) // 4
256                    elif isinstance(c, list):
257                        # 處理多模態 (text + image_url)
258                        for part in c:
259                            if part.get("type") == "text":
260                                estimated_tokens += len(part.get("text", "")) // 4
261                            elif part.get("type") == "image_url":
262                                estimated_tokens += 1000 # 預估一張圖約耗 1000 token
263                
264                await self._rate_limiter.acquire(estimated_tokens)
265
266            # 呼叫 LLM（含自動重試）
267            if not self._gateway:
268                return "❌ Engine Error: No API Gateway configured."
269
270            response = None
271            last_error: Optional[Exception] = None
272            for attempt in range(MAX_API_RETRIES + 1):
273                try:
274                    response = await self._gateway.call(
275                        messages=messages,
276                        agent_id=agent_id,
277                        stream=False,
278                    )
279                    break  # 成功，跳出重試迴圈
280                except Exception as e:
281                    last_error = e
282                    if attempt < MAX_API_RETRIES and _is_retryable(e):
283                        wait_seconds = 2 ** attempt  # 1s, 2s 指數退避
284                        logger.warning(
285                            f"⚠️ API 呼叫失敗 (重試 {attempt + 1}/{MAX_API_RETRIES})，"
286                            f"等待 {wait_seconds}s 後重試: {e}"
287                        )
288                        await asyncio.sleep(wait_seconds)
289                    else:
290                        logger.error(f"❌ API call failed (不可重試): {e}")
291                        return f"❌ API Error: {e}"
292
293            if response is None:
294                logger.error(f"❌ API call exhausted all retries: {last_error}")
295                return f"❌ API Error (重試耗盡): {last_error}"
296
297            # 解析回應
298            choice = response.get("choices", [{}])[0]
299            message = choice.get("message", {})
300            choice.get("finish_reason", "")
301
302            # Case 1: Tool calls
303            tool_calls = message.get("tool_calls", [])
304            if tool_calls:
305                # 把 assistant message (含 tool_calls) 加入歷史
306                messages.append(message)
307
308                for tc in tool_calls:
309                    func = tc.get("function", {})
310                    tool_name = func.get("name", "")
311                    try:
312                        tool_args = __import__("json").loads(func.get("arguments", "{}"))
313                    except Exception:
314                        tool_args = {}
315
316                    logger.info(f"🔧 Tool call: {tool_name}({tool_args})")
317
318                    # 發送 TOOL_CALL 事件
319                    await self.emit(EngineEvent(
320                        event_type=EventType.TOOL_CALL,
321                        payload={"tool_name": tool_name, "arguments": tool_args},
322                    ))
323
324                    # ==================================
325                    # 特殊系統工具攔截：SYS_TAKE_SCREENSHOT
326                    # ==================================
327                    if tool_name == "SYS_TAKE_SCREENSHOT":
328                        import tempfile
329                        logger.info("📸 執行原生截圖...")
330                        logger.warning("⚠️ PRIVACY CHECK: Agent is capturing the screen. Ensure no sensitive information is visible.")
331                        try:
332                            # 使用 tempfile 避免 Race Condition (Risk 3.3)
333                            fd, tmp_path = tempfile.mkstemp(suffix=".jpg")
334                            os.close(fd)
335                            
336                            # macOS native screencapture (Risk 3.6: Multi-Monitor -> use -m for main only)
337                            subprocess.run(["screencapture", "-x", "-m", "-t", "jpg", tmp_path], check=True, stderr=subprocess.DEVNULL)
338                            
339                            # 安全降維與壓縮: macOS sips 工具 (Risk 3.2: Base64 Context Explosion)
340                            # 限制最大邊長 1280px，JPEG 品質 60
341                            subprocess.run(["sips", "-Z", "1280", "-s", "formatOptions", "60", tmp_path], check=True, stdout=subprocess.DEVNULL)
342                            
343                            with open(tmp_path, "rb") as img_file:
344                                b64_img = base64.b64encode(img_file.read()).decode('utf-8')
345                                
346                            os.remove(tmp_path)
347                            
348                            # 轉換為 multimodal message format (litellm 支援的通用格式)
349                            messages.append({
350                                "role": "tool",
351                                "tool_call_id": tc.get("id", ""),
352                                "content": [
353                                    {"type": "text", "text": "Screenshot captured successfully (compressed to 1280px)."},
354                                    {
355                                        "type": "image_url",
356                                        "image_url": {
357                                            "url": f"data:image/jpeg;base64,{b64_img}"
358                                        }
359                                    }
360                                ]
361                            })
362                            
363                            await self.emit(EngineEvent(
364                                event_type=EventType.TOOL_RESULT,
365                                payload={"tool_name": tool_name, "output": "[Image Data]"},
366                            ))
367                            continue # 直接進入下一輪 LLM
368                        except Exception as e:
369                            logger.error(f"❌ 截圖失敗: {e} - 降級提供一空白圖片 (Headless mode)")
370                            # 1x1 像素的透明 PNG Base64 作為 Fallback (Headless mode)
371                            b64_dummy = "iVBORw0KGgoAAAANSUhEUgAAAAEAAAABCAYAAAAfFcSJAAAACklEQVR4nGMAAQAABQABDQottAAAAABJRU5ErkJggg=="
372                            
373                            messages.append({
374                                "role": "tool",
375                                "tool_call_id": tc.get("id", ""),
376                                "content": [
377                                    {"type": "text", "text": f"Screenshot failed (Headless or Missing Tools). Fallback dummy image provided. Error: {e}"},
378                                    {
379                                        "type": "image_url",
380                                        "image_url": {
381                                            "url": f"data:image/png;base64,{b64_dummy}"
382                                        }
383                                    }
384                                ]
385                            })
386                            
387                            await self.emit(EngineEvent(
388                                event_type=EventType.TOOL_RESULT,
389                                payload={"tool_name": tool_name, "output": "[Dummy Image Data]"},
390                            ))
391                            continue
392                            
393                    # 執行一般工具
394                    elif self._tool_executor:
395                        req = ToolCallRequest(
396                            tool_name=tool_name,
397                            arguments=tool_args,
398                        )
399                        try:
400                            result = await self._tool_executor(req)
401                        except Exception as e:
402                            result = ToolCallResult(
403                                tool_name=tool_name, success=False,
404                                output="", error=str(e),
405                            )
406                    else:
407                        result = ToolCallResult(
408                            tool_name=tool_name, success=False,
409                            output="", error="No tool executor configured",
410                        )
411
412                    # 把 tool result 加入歷史
413                    tool_output = result.output if result.success else f"Error: {result.error}"
414                    messages.append({
415                        "role": "tool",
416                        "tool_call_id": tc.get("id", ""),
417                        "content": tool_output,
418                    })
419
420                    # 發送 TOOL_RESULT 事件
421                    await self.emit(EngineEvent(
422                        event_type=EventType.TOOL_RESULT,
423                        payload={"tool_name": tool_name, "output": tool_output},
424                    ))
425
426                # 繼續 Loop（讓 LLM 看到 tool result）
427                continue
428
429            # Case 2: 純文字回覆
430            content = message.get("content", "")
431            messages.append({"role": "assistant", "content": content})
432
433            # 發送 AGENT_RESPONSE 事件
434            await self.emit(EngineEvent(
435                event_type=EventType.AGENT_RESPONSE,
436                payload={"content": content, "agent_id": agent_id},
437            ))
438
439            return content
440
441        # Watchdog: 超過最大步數
442        logger.error(f"🚨 Watchdog: 超過 {max_steps} 步，強制中止")
443        return f"⚠️ Agent 達到最大步數限制 ({max_steps})，任務中止。"
