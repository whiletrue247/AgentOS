07_PKG — GraphRAG Hybrid Retriever (v5.0 SOTA)

結合知識圖譜 (KG) 與 LLM 結構化擷取，實現：

  1. ingest_memory() — 從對話文本自動抽取 RDF 三元組存入 KG
  2. retrieve_context() — 從使用者查詢中抽取實體，召回圖譜上下文

使用 LLM Structured Output (JSON Mode) 進行真實 NER，而非 if/else 硬編碼。

 39class GraphRAG:
 40    """
 41    GraphRAG 混合檢索器 (v5.0 SOTA)。
 42    使用 LLM 做真實的 NER / Relation Extraction，不再用 if/else 假判斷。
 43    """
 44
 45    def __init__(self, engine: Any, graph: PersonalKnowledgeGraph):
 46        self.engine = engine
 47        self.graph = graph
 48        self.mem0 = Mem0Provider() if Mem0Provider else None
 49        logger.info(f"🧠 GraphRAG 混合聯想引擎初始化完成 (Mem0 Available: {bool(self.mem0)})")

 51    # ----------------------------------------------------------
 52    # 知識攝入 (Ingest)
 53    # ----------------------------------------------------------
 54    async def ingest_memory(self, session_text: str) -> int:
 55        """
 56        從對話文本中自動抽取 RDF 三元組，存入知識圖譜。
 57        使用 LLM 進行結構化 NER + Relation Extraction。
 58        """
 59        logger.info(f"📥 從對話抽取知識: {session_text[:50]}...")
 60
 61        extraction_prompt = """You are a knowledge graph extraction engine.
 62Extract ALL factual relationships from the following conversation text.
 63
 64Rules:
 65- Extract personal preferences, facts, dates, relationships, opinions
 66- Output ONLY a JSON array of triples: [["Subject", "Predicate", "Object"]]
 67- Use lowercase for all entities
 68- Be specific: "user" not "the person"
 69- Extract implicit facts too (e.g., "I hate red" → ["user", "dislikes", "red"])
 70
 71Examples:
 72- "My wife's birthday is Oct 16" → [["user's wife", "has_birthday", "october 16th"]]
 73- "I prefer dark mode" → [["user", "prefers", "dark mode"]]
 74- "Don't use red in the UI" → [["user", "dislikes", "red ui"]]
 75
 76Text to analyze:
 77"""
 78
 79        messages = [
 80            {"role": "system", "content": extraction_prompt},
 81            {"role": "user", "content": session_text},
 82        ]
 83
 84        try:
 85            response = await self.engine.gateway.call(
 86                messages=messages,
 87                agent_id="default",
 88                temperature=0.1,
 89            )
 90            raw = response["choices"][0]["message"]["content"]
 91            triplets = self._parse_triplets(raw)
 92
 93            for s, p, o in triplets:
 94                self.graph.add_triple(s, p, o, source="graphrag_extraction")
 95
 96            # 雙寫：同時存入 Mem0 向量記憶 (If available)
 97            if self.mem0:
 98                self.mem0.add_memory(session_text)
 99
100            logger.info(f"✅ 抽取並存入 {len(triplets)} 條知識三元組")
101            return len(triplets)
102
103        except Exception as e:
104            logger.error(f"❌ 知識抽取失敗: {e}")
105            return 0

110    async def retrieve_context(self, query: str) -> str:
111        """
112        從使用者查詢中抽取相關實體，然後從圖譜中召回 2-hop 子圖作為上下文。
113        """
114        logger.info(f"🔍 GraphRAG 召回: '{query[:50]}...'")
115
116        # Step 1: 使用 LLM 從查詢中抽取實體
117        entities = await self._extract_entities(query)
118
119        if not entities:
120            logger.info("ℹ️ 未從查詢中偵測到相關實體")
121            return ""
122
123        logger.info(f"🏷️ 偵測到實體: {entities}")
124
125        # Step 2: 從圖譜中召回子圖
126        subgraph = self.graph.get_subgraph(entities, max_depth=2)
127
128        if not subgraph:
129            logger.info("ℹ️ 圖譜中無相關連結")
130            return ""
131
132        # Step 3: 轉為自然語言上下文
133        context_lines = ["\n[Personal Knowledge Graph Context]:"]
134        for s, p, o in subgraph:
135            context_lines.append(f"  - {s} → {p} → {o}")
136
137        graph_str = "\n".join(context_lines)
138        
139        # Step 4: 合併 Mem0 Vector 記憶 (Hybrid Search)
140        if self.mem0:
141            mem0_str = self.mem0.search_memory(query)
142            if mem0_str:
143                graph_str += "\n" + mem0_str
144
145        logger.info(f"💡 召回 {len(subgraph)} 條知識事實 (Hybrid Mode: {bool(self.mem0)})")
146        return graph_str

151    async def _extract_entities(self, query: str) -> List[str]:
152        """使用 LLM 從查詢中抽取可能的知識圖譜實體。"""
153        prompt = """Extract entity names from the following query that might exist in a personal knowledge graph.
154Focus on: people, preferences, dates, topics, tools, colors, etc.
155
156Return ONLY a JSON array of lowercase strings. Example: ["user's wife", "dark mode", "python"]
157If no entities found, return: []
158
159Query: """
160
161        messages = [
162            {"role": "system", "content": prompt},
163            {"role": "user", "content": query},
164        ]
165
166        try:
167            response = await self.engine.gateway.call(
168                messages=messages,
169                agent_id="default",
170                temperature=0.1,
171            )
172            raw = response["choices"][0]["message"]["content"]
173            return self._parse_entity_list(raw)
174        except Exception as e:
175            logger.warning(f"⚠️ 實體抽取失敗，退回關鍵字比對: {e}")
176            return self._fallback_entity_extraction(query)

178    # ----------------------------------------------------------
179    # Fallback: 關鍵字比對 (無 LLM 時)
180    # ----------------------------------------------------------
181    @staticmethod
182    def _fallback_entity_extraction(query: str) -> List[str]:
183        """無 LLM 時的簡易實體抽取 (停用詞過濾 + 長度篩選)。"""
184        stop_words = {"the", "a", "an", "is", "are", "was", "were", "of", "in", "to",
185                      "for", "and", "or", "but", "not", "what", "who", "how", "when",
186                      "where", "do", "does", "did", "my", "your", "his", "her", "it",
187                      "this", "that", "我", "你", "的", "是", "了", "嗎", "在", "有"}
188        words = query.lower().replace("？", "").replace("?", "").split()
189        return [w for w in words if w not in stop_words and len(w) > 1]
190
191    # ----------------------------------------------------------
192    # JSON parsers
193    # ----------------------------------------------------------
194    @staticmethod
195    def _parse_triplets(raw: str) -> List[List[str]]:
196        """解析 LLM 回傳的三元組 JSON 陣列。"""
197        text = raw.strip()
198        if "```json" in text:
199            text = text.split("```json")[1].split("```")[0].strip()
200        elif "```" in text:
201            text = text.split("```")[1].split("```")[0].strip()
202        try:
203            data = json.loads(text)
204            if isinstance(data, list):
205                return [t for t in data if isinstance(t, list) and len(t) == 3]
206        except json.JSONDecodeError:
207            pass
208        return []
209
210    @staticmethod
211    def _parse_entity_list(raw: str) -> List[str]:
212        """解析 LLM 回傳的實體列表。"""
213        text = raw.strip()
214        if "```json" in text:
215            text = text.split("```json")[1].split("```")[0].strip()

222        except json.JSONDecodeError:
223            pass
224        return []
