04_Engine.streamer

04_Engine — SSE Streamer

處理 LLM API 的串流回應 (Server-Sent Events)。 負責：

  1. 解析 SSE 串流數據
  2. 將 token 片段即時轉發給訂閱者 (Messenger / Dashboard)
  3. 拼接完整回應文本
  4. 偵測 tool_call 事件

使用 Callback 模式，上層模組註冊 on_token / on_complete / on_tool_call。

 51class Streamer:
 52    """
 53    SSE 串流處理器。
 54
 55    用法：
 56        streamer = Streamer()
 57        streamer.on_token = lambda text: print(text, end="", flush=True)
 58        streamer.on_complete = lambda result: print(f"\\nDone: {len(result.full_text)} chars")
 59
 60        result = streamer.process_sse_lines(sse_lines)
 61    """
 62
 63    def __init__(self):
 64        self.on_token: Optional[OnTokenCallback] = None
 65        self.on_complete: Optional[OnCompleteCallback] = None
 66        self.on_tool_call: Optional[OnToolCallCallback] = None
 67
 68    def process_sse_lines(self, lines: list[str], provider: str = "openai") -> StreamResult:
 69        """
 70        處理一批 SSE lines (data: {...}) 並回傳彙整結果。
 71        """
 72        result = StreamResult()
 73        text_parts: list[str] = []
 74        # 累積 tool calls: { index: { id, name, args_parts } }
 75        tool_call_accum: dict[int, dict] = {}
 76
 77        for line in lines:
 78            line = line.strip()
 79            if not line or line == "data: [DONE]":
 80                continue
 81            if line.startswith("data: "):
 82                line = line[6:]
 83
 84            try:
 85                data = json.loads(line)
 86            except json.JSONDecodeError:
 87                continue
 88
 89            chunk = self._parse_chunk(data, provider)
 90            if not chunk:
 91                continue
 92
 93            # 文字片段
 94            if chunk.delta_text:
 95                text_parts.append(chunk.delta_text)
 96                if self.on_token:
 97                    self.on_token(chunk.delta_text)
 98
 99            # Tool call 片段 (逐步累積)
100            if chunk.tool_call_name is not None or chunk.tool_call_args is not None:
101                idx = 0  # 預設 index
102                if chunk.raw:
103                    # 從 raw 中取 tool_calls[0].index
104                    tc_list = chunk.raw.get("choices", [{}])[0].get("delta", {}).get("tool_calls", [])
105                    if tc_list:
106                        idx = tc_list[0].get("index", 0)
107
108                if idx not in tool_call_accum:
109                    tool_call_accum[idx] = {"id": "", "name": "", "args_parts": []}
110
111                if chunk.tool_call_id:
112                    tool_call_accum[idx]["id"] = chunk.tool_call_id
113                if chunk.tool_call_name:
114                    tool_call_accum[idx]["name"] = chunk.tool_call_name
115                if chunk.tool_call_args:
116                    tool_call_accum[idx]["args_parts"].append(chunk.tool_call_args)
117
118            # Finish reason
119            if chunk.finish_reason:
120                result.finish_reason = chunk.finish_reason
121
122        # 彙整
123        result.full_text = "".join(text_parts)
124
125        for idx in sorted(tool_call_accum.keys()):
126            tc = tool_call_accum[idx]
127            args_str = "".join(tc["args_parts"])
128            try:
129                args = json.loads(args_str) if args_str else {}
130            except json.JSONDecodeError:
131                args = {"_raw": args_str}
132
133            tool_call_dict = {
134                "id": tc["id"],
135                "name": tc["name"],
136                "arguments": args,
137            }
138            result.tool_calls.append(tool_call_dict)
139
140            if self.on_tool_call:
141                self.on_tool_call(tool_call_dict)
142
143        if self.on_complete:
144            self.on_complete(result)
145
146        return result
147
148    def _parse_chunk(self, data: dict, provider: str) -> Optional[StreamChunk]:
149        """解析單一 SSE JSON chunk"""
150
151        if provider in ("openai", "deepseek", "groq", "ollama"):
152            return self._parse_openai_chunk(data)
153        elif provider == "anthropic":
154            return self._parse_anthropic_chunk(data)
155        elif provider == "google":
156            return self._parse_google_chunk(data)
157
158        # Fallback: 嘗試 OpenAI 格式
159        return self._parse_openai_chunk(data)
160
161    def _parse_openai_chunk(self, data: dict) -> Optional[StreamChunk]:
162        choices = data.get("choices", [])
163        if not choices:
164            return None
165
166        choice = choices[0]
167        delta = choice.get("delta", {})
168        finish = choice.get("finish_reason")
169
170        chunk = StreamChunk(
171            delta_text=delta.get("content", "") or "",
172            finish_reason=finish,
173            raw=data,
174        )
175
176        # Tool calls
177        tool_calls = delta.get("tool_calls", [])
178        if tool_calls:
179            tc = tool_calls[0]
180            chunk.tool_call_id = tc.get("id")
181            func = tc.get("function", {})
182            chunk.tool_call_name = func.get("name")
183            chunk.tool_call_args = func.get("arguments")
184
185        return chunk
186
187    def _parse_anthropic_chunk(self, data: dict) -> Optional[StreamChunk]:
188        event_type = data.get("type", "")
189
190        if event_type == "content_block_delta":
191            delta = data.get("delta", {})
192            if delta.get("type") == "text_delta":
193                return StreamChunk(delta_text=delta.get("text", ""))
194            if delta.get("type") == "input_json_delta":
195                return StreamChunk(tool_call_args=delta.get("partial_json", ""))
196
197        if event_type == "content_block_start":
198            block = data.get("content_block", {})
199            if block.get("type") == "tool_use":
200                return StreamChunk(
201                    tool_call_id=block.get("id"),
202                    tool_call_name=block.get("name"),
203                )
204
205        if event_type == "message_delta":
206            return StreamChunk(finish_reason=data.get("delta", {}).get("stop_reason"))
207
208        return None
209
210    def _parse_google_chunk(self, data: dict) -> Optional[StreamChunk]:
211        candidates = data.get("candidates", [])
212        if not candidates:
213            return None
214
215        content = candidates[0].get("content", {})
216        parts = content.get("parts", [])
217        if not parts:
218            return None
219
220        text = parts[0].get("text", "")
221        finish = candidates[0].get("finishReason")
222
223        return StreamChunk(delta_text=text, finish_reason=finish)
