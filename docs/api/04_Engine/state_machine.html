04_Engine — 任務狀態機 + Checkpoint

追蹤每個任務的生命週期狀態。 支援 Checkpoint 機制：任務中斷後可從最近的 Checkpoint 恢復。

狀態流轉： PENDING → RUNNING → COMPLETED → FAILED → PAUSED (等待人類回覆) → CANCELLED

  1"""
  204_Engine — 任務狀態機 + Checkpoint
  3===================================
  4追蹤每個任務的生命週期狀態。
  5支援 Checkpoint 機制：任務中斷後可從最近的 Checkpoint 恢復。
  6
  7狀態流轉：
  8  PENDING → RUNNING → COMPLETED
  9                    → FAILED
 10                    → PAUSED (等待人類回覆)
 11                    → CANCELLED

 68class StateMachine:
 69    """任務狀態機"""
 70
 71    def __init__(self, checkpoint_dir: Optional[str] = None):
 72        if checkpoint_dir is None:
 73            checkpoint_dir = str(get_data_dir() / "checkpoints")
 74        self._tasks: dict[str, TaskRecord] = {}
 75        self._checkpoint_dir = Path(checkpoint_dir)
 76        self._checkpoint_dir.mkdir(parents=True, exist_ok=True)
 77
 78    # ========================================
 79    # 生命週期
 80    # ========================================
 81
 82    def create_task(self, description: str, agent_id: str = "default", max_steps: int = 50) -> TaskRecord:
 83        """建立新任務"""
 84        task = TaskRecord(
 85            task_id=str(uuid.uuid4())[:8],
 86            description=description,
 87            state=TaskState.PENDING.value,
 88            agent_id=agent_id,
 89            created_at=datetime.now().isoformat(),
 90            max_steps=max_steps,
 91        )
 92        self._tasks[task.task_id] = task
 93        logger.info(f"📋 任務建立: [{task.task_id}] {description}")
 94        return task
 95
 96    def start(self, task_id: str) -> TaskRecord:
 97        task = self._get(task_id)
 98        task.state = TaskState.RUNNING.value
 99        task.started_at = datetime.now().isoformat()
100        logger.info(f"▶️ 任務開始: [{task_id}]")
101        return task
102
103    def complete(self, task_id: str, result: str = "") -> TaskRecord:
104        task = self._get(task_id)
105        task.state = TaskState.COMPLETED.value
106        task.completed_at = datetime.now().isoformat()
107        task.result = result
108        self._cleanup_checkpoint(task_id)
109        logger.info(f"✅ 任務完成: [{task_id}]")
110        return task
111
112    def fail(self, task_id: str, error: str = "") -> TaskRecord:
113        task = self._get(task_id)
114        task.state = TaskState.FAILED.value
115        task.completed_at = datetime.now().isoformat()
116        task.error = error
117        logger.error(f"❌ 任務失敗: [{task_id}] {error}")
118        return task
119
120    def pause(self, task_id: str) -> TaskRecord:
121        """暫停任務（等待人類回覆）"""
122        task = self._get(task_id)
123        task.state = TaskState.PAUSED.value
124        logger.info(f"⏸️ 任務暫停: [{task_id}] (waiting for human)")
125        return task
126
127    def resume(self, task_id: str) -> TaskRecord:
128        """恢復暫停的任務"""
129        task = self._get(task_id)
130        task.state = TaskState.RUNNING.value
131        logger.info(f"▶️ 任務恢復: [{task_id}]")
132        return task
133
134    def cancel(self, task_id: str) -> TaskRecord:
135        task = self._get(task_id)
136        task.state = TaskState.CANCELLED.value
137        task.completed_at = datetime.now().isoformat()
138        self._cleanup_checkpoint(task_id)
139        logger.info(f"🚫 任務取消: [{task_id}]")
140        return task
141
142    def update_step(self, task_id: str, step: int) -> None:
143        task = self._get(task_id)
144        task.current_step = step
145
146    # ========================================
147    # Checkpoint
148    # ========================================
149
150    def save_checkpoint(self, task_id: str, messages: list[dict], step: int) -> None:
151        """儲存對話歷史快照"""
152        task = self._get(task_id)
153        task.checkpoint_messages = messages
154        task.checkpoint_step = step
155
156        cp_path = self._checkpoint_dir / f"{task_id}.json"
157        with open(cp_path, "w", encoding="utf-8") as f:
158            json.dump({
159                "task_id": task_id,
160                "step": step,
161                "messages": messages,
162                "state": task.state,
163            }, f, ensure_ascii=False, indent=2)
164
165        logger.debug(f"💾 Checkpoint 已存: [{task_id}] step={step}")
166
167    def load_checkpoint(self, task_id: str) -> Optional[dict]:
168        """載入 Checkpoint"""
169        cp_path = self._checkpoint_dir / f"{task_id}.json"
170        if not cp_path.exists():
171            return None
172
173        with open(cp_path, "r", encoding="utf-8") as f:
174            data = json.load(f)
175
176        logger.info(f"📂 Checkpoint 已載入: [{task_id}] step={data.get('step')}")
177        return data
178
179    def _cleanup_checkpoint(self, task_id: str) -> None:
180        cp_path = self._checkpoint_dir / f"{task_id}.json"
181        if cp_path.exists():
182            cp_path.unlink()
183
184    # ========================================
185    # 查詢
186    # ========================================
187
188    def get_task(self, task_id: str) -> Optional[TaskRecord]:
189        return self._tasks.get(task_id)
190
191    def list_tasks(self, state: Optional[str] = None) -> list[TaskRecord]:
192        tasks = list(self._tasks.values())
193        if state:
194            tasks = [t for t in tasks if t.state == state]
195        return tasks
196
197    def _get(self, task_id: str) -> TaskRecord:
198        task = self._tasks.get(task_id)
199        if not task:
200            raise ValueError(f"Task not found: {task_id}")
201        return task
